cd					- перемещение в другие директории
cd ~				- переместиться в домашнюю директорию (папку пользователя C:/Users/user)
cd ..				- переместиться в родительскую директории (на одну папку выше)

ls					- просмотр файлов
ls -a				- просмотр всех файлов, включая скрытые
pwd					- просмотр пути текущей директории
mkdir 				- создать новые папки
touch 				- создать файлы
cp <что-копировать> <что-копировать> <куда-копировать>		- скопировать файлы/папки в заданную директорию
mv <что-переносить> <что-переносить> <куда-переносить>		- перенести файлы/папки в заданную директорию
cat 				- прочесть содержимое файла
rm 					- удалить файл
rmdir 				- удалить папку
rm -r 				- удалить папку со всем его содержимым

&& 					- разделитель между двумя командами в одной строке (то есть сразу выполнить две команды A && B)

echo 'Сообщение'	- Выводит "Сообщение" в командную строку

echo 'Сообщение' >> file.txt		- дописывает в конец файла file.txt "Сообщение"
echo 'Сообщение' > file.txt 		- перепишет содержимое файла file.txt на "Сообщение"
